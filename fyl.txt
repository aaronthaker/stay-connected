import os
import pandas as pd

# Directory paths
hypothesis_dir = 'path_to_your_directory/Hypothesis_Seeds'
reference_dir = 'path_to_your_directory/Reference_Seeds'

# Collecting filenames and their contents
data = []
for filename in os.listdir(hypothesis_dir):
    if filename.endswith('_Hypothesis.txt'):
        hypothesis_path = os.path.join(hypothesis_dir, filename)
        # Extract the ID part which can include underscores
        file_id = filename.rsplit('_', 1)[0]
        
        # Corresponding reference file
        reference_filename = f"{file_id}_Reference.txt"
        reference_path = os.path.join(reference_dir, reference_filename)

        # Read content from hypothesis file
        with open(hypothesis_path, 'r', encoding='utf-8') as file:
            hypothesis_content = file.read()

        # Read content from reference file
        try:
            with open(reference_path, 'r', encoding='utf-8') as file:
                reference_content = file.read()
        except FileNotFoundError:
            reference_content = "No corresponding file found"

        # Append to data list
        data.append({'Filename': filename, 'Hypothesis Content': hypothesis_content, 'Reference Content': reference_content})

# Create a DataFrame and save to CSV
df = pd.DataFrame(data)
df.to_csv('output.csv', index=False)
